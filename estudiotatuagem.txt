CREATE TABLE senioridade 
( 
 idsenioridade 	INT 		AUTO_INCREMENT PRIMARY KEY,  
 nome 		VARCHAR(100) 	NOT NULL 
); 

CREATE TABLE tatuador
( 
 idtatuador 	 INT 		AUTO_INCREMENT PRIMARY KEY,  
 codsenioridade  INT		NOT NULL,  
 nometatuador 	 VARCHAR()	NOT NULL,  
 telefone 	 VARCHAR()	NULL,  
 cpf 		 CHAR		NOT NULL UNIQUE,  
 datacontratacao DATE		NOT NULL,  	
 email	 	 VARCHAR(150)	NULL UNIQUE,  
 FOREIGN KEY (codsenioridade) REFERENCES senioridade (codsenioridade)
); 

CREATE TABLE Estilo Tatuagem 
( 
 idEstilo 	INT 		AUTO_INCREMENT PRIMARY KEY,   
 Nome 		VACHAR		NOT NULL UNIQUE,  
); 

CREATE TABLE Cliente 
( 
 idcliente 	INT 		AUTO_INCREMENT PRIMARY KEY,  
 Telefone 	INT		NULL,  
 Nome 		VARCHAR(255)	NOT NULL UNIQUE,  
 cpf 		CHAR		NOT NULL,  
 DataN 		INT		NOT NULL,  
); 



CREATE TABLE agendamento 
( 
 idCliente 	INT		AUTO_INCREMENT PRIMARY KEY,  
 idTatuador 	INT		AUTO_INCREMENT PRIMARY KEY,  
 Data 		DATE		NOT NULL,  
 Horario 	DATE		NOT NULL,  
 Descricao 	VARCHAR(255)	NULL,  
 Valor 		FLOAT		NOT NULL,  
 Status 	VARCHAR(255)	NOT NULL,  
); 

CREATE TABLE Domina 
( 
 idtatuador 	INT 	AUTO_INCREMENT PRIMARY KEY,  
 idEstilo 	INT 	AUTO_INCREMENT PRIMARY KEY,  
);
