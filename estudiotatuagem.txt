CREATE TABLE senioridade 
( 
 idsenioridade 	INT 		AUTO_INCREMENT PRIMARY KEY,  
 nome 		VARCHAR(100) 	NOT NULL 
); 

CREATE TABLE tatuador
( 
 idtatuador 	 INT 	AUTO_INCREMENT PRIMARY KEY,  
 codsenioridade  INT,  
 nometatuador 	 VARCHAR(),  
 telefone 	 VARCHAR(),  
 cpf 		 CHAR,  
 datacontratacao DATE,  
 email	 	 VARCHAR(150),  
); 

CREATE TABLE Estilo Tatuagem 
( 
 idEstilo INT PRIMARY KEY,  
 Descricao INT,  
 Nome INT,  
); 

CREATE TABLE Cliente 
( 
 idcliente INT PRIMARY KEY,  
 Telefone INT,  
 Nome INT,  
 cpf INT,  
 DataN INT,  
); 



CREATE TABLE agendamento 
( 
 idCliente INT,  
 idTatuador INT,  
 Data INT,  
 Horario INT,  
 Descricao INT,  
 Valor INT,  
 Status INT,  
); 

CREATE TABLE Domina 
( 
 idtatuador INT PRIMARY KEY,  
 idEstilo INT PRIMARY KEY,  
); 

ALTER TABLE Tatuador ADD FOREIGN KEY(idsenioridade) REFERENCES senioridade (idsenioridade)
ALTER TABLE agendamento ADD FOREIGN KEY(idCliente) REFERENCES Cliente (idCliente)
ALTER TABLE agendamento ADD FOREIGN KEY(idTatuador) REFERENCES Tatuador (idTatuador)
ALTER TABLE Domina ADD FOREIGN KEY(idtatuador) REFERENCES Tatuador (idtatuador)
ALTER TABLE Domina ADD FOREIGN KEY(idEstilo) REFERENCES Estilo Tatuagem (idEstilo)
